const express = require('express');
const router = express.Router();
const { Parser } = require('json2csv');
const pdf = require('pdfkit');
const fs = require('fs');
const Match = require('../models/Match');
const Event = require('../models/Event');
const User = require('../models/User');
 
// Generate Volunteer Participation Report
router.get('/volunteer', async (req, res) => {
  const { format } = req.query;
 
  try {
    // Fetch volunteer participation data
    const matches = await Match.find()
      .populate('userId', 'name email')
      .populate('eventId', 'eventName eventDate location');
 
    const data = matches.map((match) => ({
      volunteerName: match.userId?.name || 'N/A',
      volunteerEmail: match.userId?.email || 'N/A',
      eventName: match.eventId?.eventName || 'N/A',
      eventDate: match.eventId?.eventDate || 'N/A',
      eventLocation: match.eventId?.location || 'N/A',
    }));
 
    if (format === 'csv') {
      // Convert to CSV
      const fields = ['volunteerName', 'volunteerEmail', 'eventName', 'eventDate', 'eventLocation'];
      const json2csvParser = new Parser({ fields });
      const csv = json2csvParser.parse(data);
 
      res.header('Content-Type', 'text/csv');
      res.attachment('volunteer_report.csv');
      return res.send(csv);
    } else if (format === 'pdf') {
      // Generate PDF
      const doc = new pdf();
      const filePath = './volunteer_report.pdf';
      doc.pipe(fs.createWriteStream(filePath));
 
      doc.fontSize(18).text('Volunteer Participation Report', { align: 'center' });
      doc.moveDown();
 
      data.forEach((item) => {
        doc
          .fontSize(12)
          .text(`Volunteer Name: ${item.volunteerName}`)
          .text(`Email: ${item.volunteerEmail}`)
          .text(`Event: ${item.eventName}`)
          .text(`Date: ${new Date(item.eventDate).toLocaleDateString()}`)
          .text(`Location: ${item.eventLocation}`)
          .moveDown();
      });
 
      doc.end();
 
      doc.on('finish', () => {
        res.download(filePath, () => {
          fs.unlinkSync(filePath);
        });
      });
    } else {
      return res.status(400).json({ message: 'Invalid format. Use "csv" or "pdf".' });
    }
  } catch (error) {
    console.error('Error generating volunteer report:', error);
    res.status(500).json({ message: 'Error generating volunteer report.' });
  }
});
 
// Generate Event Management Report
router.get('/events', async (req, res) => {
  const { format } = req.query;
 
  try {
    // Fetch event data with volunteer counts
    const events = await Event.find().lean();
    const eventData = await Promise.all(
      events.map(async (event) => {
        const volunteerCount = await Match.countDocuments({ eventId: event._id });
        return {
          eventName: event.eventName,
          eventDate: event.eventDate,
          location: event.location,
          volunteerCount,
        };
      })
    );
 
    if (format === 'csv') {
      // Convert to CSV
      const fields = ['eventName', 'eventDate', 'location', 'volunteerCount'];
      const json2csvParser = new Parser({ fields });
      const csv = json2csvParser.parse(eventData);
 
      res.header('Content-Type', 'text/csv');
      res.attachment('event_report.csv');
      return res.send(csv);
    } else if (format === 'pdf') {
      // Generate PDF
      const doc = new pdf();
      const filePath = './event_report.pdf';
      doc.pipe(fs.createWriteStream(filePath));
 
      doc.fontSize(18).text('Event Management Report', { align: 'center' });
      doc.moveDown();
 
      eventData.forEach((event) => {
        doc
          .fontSize(12)
          .text(`Event Name: ${event.eventName}`)
          .text(`Date: ${new Date(event.eventDate).toLocaleDateString()}`)
          .text(`Location: ${event.location}`)
          .text(`Volunteer Count: ${event.volunteerCount}`)
          .moveDown();
      });
 
      doc.end();
 
      doc.on('finish', () => {
        res.download(filePath, () => {
          fs.unlinkSync(filePath);
        });
      });
    } else {
      return res.status(400).json({ message: 'Invalid format. Use "csv" or "pdf".' });
    }
  } catch (error) {
    console.error('Error generating event report:', error);
    res.status(500).json({ message: 'Error generating event report.' });
  }
});
 
module.exports = router;
